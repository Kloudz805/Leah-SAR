# --- Проєкт Leah: Пресет для пошуку в завалах (SAR у приміщенні) ---
# Призначення: Низька чутливість, висока стабільність, увімкнено режим відновлення після зіткнення.

# 1. МАСШТАБУВАННЯ ГАЗУ (Точність / Precision)
# Обмежує потужність до 60% для точного контролю висоти у приміщеннях.
set throttle_limit_type = SCALE
set throttle_limit_percent = 60

# 2. ВІДНОВЛЕННЯ ПІСЛЯ ЗІТКНЕННЯ (Безпека / Crash Recovery)
# Якщо дрон зачепить стіну, він спробує вирівнятися замість того, щоб впасти (disarm).
set crash_recovery = ON

# 3. ПЛАВНІ РЕЙТИ (Фільтр "Адреналіну")
# Знижує чутливість стіків, щоб запобігти різким рухам через тремтіння рук.
set roll_rc_rate = 50
set pitch_rc_rate = 50
set yaw_rc_rate = 50
set roll_expo = 40
set pitch_expo = 40
set yaw_expo = 30

# 4. НАВІГАЦІЯ
# Готує дрон до роботи в режимах Angle/Horizon (стабілізація).
set small_angle = 180
save

# ---------------------------------------------------
# ВАЖЛИВІ РУЧНІ НАЛАШТУВАННЯ (Зробити окремо)
# ---------------------------------------------------

♠ Налаштування ESC (Крутний момент двигуна)
Критичний крок: У завалах вам потрібне «зчеплення» і сила, а не плавність.
• Відкрийте BLHeliSuite32 або ESC-Configurator. Зчитайте налаштування (Read Settings).
• Змініть частоту PWM (PWM Frequency) на **24kHz**.
• Чому? 24kHz забезпечує максимальний крутний момент на низах. Це запобігає «зриву» (washout) дрона в тісних кутах або турбулентному повітрі.
• Збережіть налаштування (Write/Save).

♠ Калібрування батареї (Безпека)
Критичний крок: Не літайте наосліп. Ви повинні точно знати, коли повертатися.
• Підключіть повністю заряджену батарею.
• Перейдіть на вкладку "Power & Battery" у Betaflight.
• Порівняйте значення "Voltage" (Вольтаж) з реальним напругою батареї (наприклад, 16.8V для 4S, 25.2V для 6S).
• Налаштовуйте "Scale" (Масштаб), доки значення не будуть ідеально збігатися.
• Збережіть (Save).

♠ Цілісність Сигналу (Важливо для бетону/завалів)
• Потужність VTX: Встановіть на **МАКСИМУМ** (рекомендовано 800mW або >1W). Низька потужність = втрата відео за стінами.
• OSD: Увімкніть **Link Quality (LQ)** та **RSSI dBm**. Не покладайтеся лише на відсоток RSSI.
• Увага: Якщо LQ падає нижче 70, негайно повертайтеся.
